\section{Bayesian Linear Regression}
% =============================================================================

\sectionslide{Bayesian Linear Regression}{%
  OLS gives a single coefficient estimate.
  Bayesian regression gives a full distribution over every coefficient,
  quantifying exactly how uncertain we are about each effect.}

% --- Slide: OLS vs Bayesian regression ---------------------------------------
\begin{frame}{OLS vs.\ Bayesian Regression: What Changes?}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{OLS (Lectures 2 and 8):}
      \[
        \hat{\boldsymbol{\beta}} = (\mathbf{X}^{\top}\mathbf{X})^{-1}\mathbf{X}^{\top}\mathbf{y}
      \]
      \begin{itemize}\footnotesize
        \item Single point estimate $\hat{\boldsymbol{\beta}}$
        \item Standard errors derived from asymptotic theory
        \item Confidence intervals rely on large-sample normality
        \item Regularization (Ridge, LASSO) is an ad-hoc fix
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{Bayesian regression:}
      \[
        p(\boldsymbol{\beta} \mid \mathbf{y}) \propto p(\mathbf{y} \mid \boldsymbol{\beta}) \cdot p(\boldsymbol{\beta})
      \]
      \begin{itemize}\footnotesize
        \item Full posterior distribution for each $\beta_j$
        \item Credible intervals are exact probability statements
        \item Regularization = informative prior
              (Normal $\equiv$ Ridge; Laplace $\equiv$ LASSO)
        \item Works even with small samples
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.1cm}
  \begin{keybox}
    \small Bayesian regression answers questions OLS cannot:
    ``What is the probability that advertising elasticity exceeds 0.5?''
    or ``Did price sensitivity increase or decrease?''
  \end{keybox}
\end{frame}

% --- Slide: Bayesian regression in PyMC ------------------------------------
\begin{frame}{Bayesian Linear Regression in PyMC}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{examplebox}{Retail Sales Regression}
        {\scriptsize
        \texttt{with pm.Model() as blr:}\\
        \texttt{  intercept = pm.Normal('intercept',}\\
        \texttt{    mu=y.mean(), sigma=y.std())}\\
        \texttt{  beta = pm.Normal('beta',}\\
        \texttt{    mu=0, sigma=1, shape=X.shape[1])}\\
        \texttt{  sigma = pm.HalfNormal('sigma', y.std())}\\[1pt]
        \texttt{  mu = intercept + X @ beta}\\[1pt]
        \texttt{  pm.Normal('y\_obs', mu, sigma, observed=y)}\\[1pt]
        \texttt{  trace = pm.sample(2000)}\\[1pt]
        \texttt{pm.plot\_posterior(trace, var\_names=['beta'])}
        }
      \end{examplebox}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{Standardize predictors first:}
      \begin{itemize}\footnotesize
        \item Center and scale: \texttt{X\_std = (X - X.mean()) / X.std()}
        \item With standardized inputs, $\mathcal{N}(0,1)$ priors are
              weakly informative
        \item Coefficients = ``effect of 1-SD increase in $x_j$ on $y$''
      \end{itemize}
      \vspace{0.1cm}
      \begin{keybox}
        \small A Normal prior on $\beta$ is equivalent to Ridge
        regularization. A Laplace prior is equivalent to LASSO.
        Bayesian inference subsumes regularization.
      \end{keybox}
    \end{column}
  \end{columns}
\end{frame}

\section{Interpreting Posterior Coefficient Distributions}
% =============================================================================

\sectionslide{Interpreting Posterior Coefficient Distributions}{%
  Credible intervals, probability of direction, and
  conditional effect analysis.}

% --- Slide: Reading posterior plots -----------------------------------------
\begin{frame}{Reading Posterior Distributions}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{For each $\beta_j$, the posterior gives:}
      \begin{description}\footnotesize
        \item[Posterior mean] Best single estimate
        \item[Posterior median] More robust if distribution is skewed
        \item[HDI] Highest Density Interval: shortest interval
              containing $X$\% of the probability mass
        \item[Prob.\ of direction] $P(\beta_j > 0)$ or $P(\beta_j < 0)$ ---
              more natural than a $p$-value
        \item[ROPE] Region Of Practical Equivalence: fraction of the
              posterior in $|\beta_j| < \delta$ (``negligible effect'').
              High ROPE mass = statistically present but practically irrelevant.
      \end{description}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{examplebox}{Business Example}
        \small Posterior for advertising elasticity $\beta_{\text{ads}}$:
        \begin{itemize}\footnotesize
          \item Posterior mean: 0.32
          \item 94\% HDI: [0.18, 0.47]
          \item $P(\beta_{\text{ads}} > 0) = 0.998$
        \end{itemize}
        \textbf{Interpretation:} ``99.8\% probability of a positive
        effect. A 10\% increase in ads raises sales by 3.2\%
        (credible range: 1.8\%--4.7\%).''
      \end{examplebox}
    \end{column}
  \end{columns}
\end{frame}

\section{Directed Acyclic Graphs and Causal Structure}
% =============================================================================

\sectionslide{DAGs and Causal Structure}{%
  A Directed Acyclic Graph makes the assumed causal structure explicit,
  helping decide which variables to include in a model.}

% --- Slide: DAGs -------------------------------------------------------------
\begin{frame}{Directed Acyclic Graphs (DAGs)}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      A \textbf{DAG} is a graph where:
      \begin{itemize}\footnotesize
        \item \textbf{Nodes} = variables
        \item \textbf{Directed arrows} = causal relationships ($A \to B$: $A$ causes $B$)
        \item \textbf{Acyclic} = no variable causes itself
      \end{itemize}
      \vspace{0.05cm}
      \textbf{How to build:} For each variable pair, ask ``Could $A$ directly cause $B$?''
      Draw an arrow if yes; direction = causation, not correlation.

      \vspace{0.05cm}
      \textbf{Why use DAGs?}
      \begin{itemize}\footnotesize
        \item Makes causal structure explicit before modeling
        \item Identifies \textbf{confounders} (must control for)
        \item Identifies \textbf{colliders} (must NOT control for)
        \item Clarifies which coefficients are causal vs.\ associational
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{examplebox}{Retail Sales DAG}
        \footnotesize
        \begin{itemize}\footnotesize
          \item Advertising $\to$ Sales
          \item Price $\to$ Sales
          \item Season $\to$ Advertising
          \item Season $\to$ Sales
        \end{itemize}
        \textbf{Season is a confounder:}
        omitting it causes the advertising coefficient
        to absorb seasonal variation --- biased upward.
      \end{examplebox}
    \end{column}
  \end{columns}
\end{frame}

\section{Conditional Analysis and Group-Specific Models}
% =============================================================================

\sectionslide{Conditional Analysis and Group-Specific Models}{%
  Use posterior samples to ask conditional questions:
  what happens to the forecast if this variable changes?}

% --- Slide: Posterior predictive for scenarios --------------------------------
\begin{frame}{Scenario Analysis with Posterior Samples}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \small\textbf{Procedure:}
      \begin{enumerate}\footnotesize
        \item Draw $S$ posterior samples $\boldsymbol{\beta}^{(1)}, \ldots, \boldsymbol{\beta}^{(S)}$
        \item For scenario $\mathbf{x}^*$, compute
              $\hat{y}^{(s)} = \mathbf{x}^{*\top}\boldsymbol{\beta}^{(s)}$
        \item The distribution of $\hat{y}^{(s)}$ is the
              \textbf{posterior predictive distribution}
      \end{enumerate}
      \vspace{0.1cm}
      \begin{examplebox}{``What if we raise price by 5\%?''}
        {\scriptsize
        \texttt{x\_new = x\_base.copy()}\\
        \texttt{x\_new['price'] *= 1.05}\\
        \texttt{b = trace.posterior['beta'].values}\\
        \texttt{y\_draws = b @ x\_new}\\
        \texttt{print(f"E[sales]: \{y\_draws.mean():.0f\}")}\\
        \texttt{print(f"95\% CI: \{np.percentile(y\_draws,[2.5,97.5])\}")}
        }
      \end{examplebox}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{Group-specific models:}
      \begin{itemize}\footnotesize
        \item Hierarchical structure (L11) gives
              \textbf{store-specific} posteriors
        \item Scenario effects differ by store:
              high-elasticity vs.\ low-elasticity
        \item Posterior uncertainty reveals which
              stores drive total revenue risk
      \end{itemize}
      \vspace{0.1cm}
      \begin{keybox}
        \small \textbf{Decision analysis:}
        choose the action (price, promo, budget)
        that maximizes expected profit given
        the full posterior over outcomes.
      \end{keybox}
    \end{column}
  \end{columns}
\end{frame}

% --- Slide: Key takeaways ---------------------------------------------------
\begin{frame}{Lecture 12: Key Takeaways}
  \begin{keybox}
    {\small
    \begin{enumerate}
      \item \textbf{Bayesian linear regression} places priors on coefficients
            and returns a full posterior distribution for each $\beta_j$.
            A Normal prior is equivalent to Ridge regularization;
            a Laplace prior is equivalent to LASSO.

      \item Posterior coefficient plots give:
            posterior mean, HDI, and $P(\beta_j > 0)$ ---
            richer information than a single $p$-value.

      \item \textbf{DAGs} make the causal structure explicit,
            helping identify confounders (must include) and
            colliders (must exclude). Always draw the DAG before
            specifying a regression model.

      \item \textbf{Scenario analysis} with posterior samples:
            plug any counterfactual $\mathbf{x}^*$ into posterior draws
            to get a full predictive distribution for that scenario.

      \item \textbf{No single method is best for all forecasting problems.}
            Build intuition for when to use each tool in this course's toolkit,
            and combine methods (e.g., point forecast + Bayesian uncertainty)
            when the decision stakes are high.
    \end{enumerate}
    }
  \end{keybox}
\end{frame}
