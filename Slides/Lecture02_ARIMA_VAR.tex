\section{Stationarity and Differencing}
% =============================================================================

\sectionslide{Stationarity and Differencing}{%
  Before fitting any model, the series must have a stable mean and variance.}

% --- Slide: Weak stationarity -------------------------------------------------
\begin{frame}{What Is Stationarity?}
  \begin{definitionbox}{Weak (Covariance) Stationarity}
    A time series $\{y_t\}$ is \textbf{weakly stationary} if:
    \begin{enumerate}\small
      \item $\E[y_t] = \mu$ \quad (constant mean over time)
      \item $\Var(y_t) = \sigma^2 < \infty$ \quad (constant, bounded variance)
      \item $\Cov(y_t, y_{t-k}) = \gamma_k$ depends only on lag $k$, not on $t$
    \end{enumerate}
  \end{definitionbox}
  \vspace{0.1cm}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Why it matters:}
      \begin{itemize}\small
        \item If the mean drifts, no single value is forecastable
        \item If variance grows, prediction intervals become infinite
        \item Stationarity makes the past informative about the future
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{warningbox}
        \textbf{Common culprits:}
        \begin{itemize}\small
          \item Trending series (GDP, prices, sales)
          \item Variance that grows with level
        \end{itemize}
        \textbf{Fix:} first-differencing removes a stochastic trend.
      \end{warningbox}
    \end{column}
  \end{columns}
  \muted{\footnotesize\itshape
    Rule of thumb: visible trend or growing variance $\Rightarrow$
    likely non-stationary. Test formally with ADF or KPSS.}
\end{frame}

% --- Slide: Unit root + differencing ------------------------------------------
\begin{frame}{Unit Roots and Differencing}
  \textbf{AR(1) model:} $y_t = \phi_1\,y_{t-1} + \varepsilon_t$
  \begin{center}
    {\small
    \begin{tabular}{lll}
      \toprule
      \textbf{Condition} & \textbf{Behavior} & \textbf{Implication} \\
      \midrule
      $|\phi_1| < 1$ & Shock decays, reverts to mean & Stationary AR(1) \\
      $\phi_1 = 1$   & Shock persists \emph{permanently} & Random walk --- unit root \\
      \bottomrule
    \end{tabular}
    }
  \end{center}
  \vspace{0.1cm}
  \textbf{First differencing} removes a stochastic trend:
  \[
    \Delta y_t = y_t - y_{t-1}
  \]
  \textbf{Testing:} \textbf{ADF test} ($H_0$: unit root) and \textbf{KPSS test}
  ($H_0$: stationary). Use both; they test opposite nulls.
  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{keybox}
        {\small $d=0$: already stationary.
        $d=1$: one first difference.
        $D=1$: one seasonal difference ($\Delta_{12}y_t = y_t - y_{t-12}$).}
      \end{keybox}
    \column{0.48\textwidth}
      \begin{warningbox}
        \textbf{Over-differencing} induces negative autocorrelation.
        Apply unit-root test \emph{after} differencing; stop at $d=1$
        if the result is stationary.
      \end{warningbox}
  \end{columns}
\end{frame}

% --- Slide: ACF and PACF for identification -----------------------------------
\begin{frame}{ACF and PACF: Model Identification}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Autocorrelation function (ACF):}
      $\rho_k = \mathrm{Corr}(y_t, y_{t-k})$
      \begin{itemize}\small
        \item MA($q$): \pos{cuts off} after lag $q$
        \item AR($p$): tails off geometrically
      \end{itemize}
      \vspace{0.2cm}
      \textbf{Partial autocorrelation (PACF):}
      correlation between $y_t$ and $y_{t-k}$ controlling for intermediate lags
      \begin{itemize}\small
        \item AR($p$): \pos{cuts off} after lag $p$
        \item MA($q$): tails off geometrically
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{center}
        {\small
        \begin{tabular}{lll}
          \toprule
          \textbf{Model} & \textbf{ACF} & \textbf{PACF} \\
          \midrule
          AR($p$)  & Tails off & Cuts off at $p$ \\
          MA($q$)  & Cuts off at $q$ & Tails off \\
          ARMA     & Tails off & Tails off \\
          RW / I(1) & Decays slowly & Big spike at 1 \\
          \bottomrule
        \end{tabular}
        }
      \end{center}
      \begin{keybox}
        \small Always unit-root test and difference
        \emph{before} reading ACF/PACF.
        The ACF of a non-stationary series is not interpretable.
      \end{keybox}
    \end{column}
  \end{columns}
\end{frame}

\section{ARIMA Models}
% =============================================================================

\sectionslide{ARIMA Models}{%
  Combine differencing with AR and MA components into one unified model.}

% --- Slide: ARIMA definition --------------------------------------------------
\begin{frame}{ARIMA($p$,$d$,$q$)}
  \begin{definitionbox}{ARIMA($p$,$d$,$q$) Model}
    Apply an ARMA($p$,$q$) to the $d$-times differenced series:
    \[
      y_t = c + \sum_{i=1}^{p}\phi_i\,y_{t-i}
              + \varepsilon_t
              + \sum_{j=1}^{q}\theta_j\,\varepsilon_{t-j}
      \quad \text{after differencing $d$ times}
    \]
  \end{definitionbox}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Parameters:}
      \begin{itemize}\small
        \item $p$: AR lags (past values)
        \item $d$: number of differences needed
        \item $q$: MA lags (past errors)
      \end{itemize}
      \textbf{Key special cases:}
      \begin{itemize}\small
        \item ARIMA(0,1,0): random walk
        \item ARIMA(0,1,1): equivalent to SES
        \item ARIMA(1,0,0): AR(1)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{examplebox}{Numeric Example}
        ARIMA(1,1,0) with $\hat{\phi}_1=0.5$:\\
        $y_{100}=120$, $y_{99}=118$\\
        $\Delta y_{100} = 2$\\[3pt]
        Forecast:
        $\hat{y}_{101|100} = 120 + 0.5 \times 2 = \mathbf{121}$
      \end{examplebox}
      \begin{keybox}
        \small
        \textbf{Python:}
        \texttt{pmdarima.auto\_arima()} searches automatically.
        \texttt{statsmodels.SARIMAX} fits any specified model.
      \end{keybox}
    \end{column}
  \end{columns}
\end{frame}

% --- Slide: SARIMA ------------------------------------------------------------
\begin{frame}{SARIMA: Adding Seasonal Terms}
  For monthly or quarterly data, autocorrelation often appears at lags
  $m, 2m, 3m, \ldots$ SARIMA($p$,$d$,$q$)($P$,$D$,$Q$)[$m$] adds seasonal
  AR and MA polynomials:
  \begin{keybox}
    Think of SARIMA as two ARIMA models operating at different time scales:
    \begin{itemize}\small
      \item Non-seasonal part $(p,d,q)$: captures week-to-week dynamics
      \item Seasonal part $(P,D,Q)[m]$: captures year-over-year patterns at lag $m$
    \end{itemize}
  \end{keybox}
  \vspace{0.1cm}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Practical workflow:}
      \begin{enumerate}\small
        \item ADF/KPSS $\to$ choose $d$
        \item Seasonal unit root test $\to$ choose $D$
        \item ACF/PACF on differenced series $\to$ guide $p,q,P,Q$
        \item Minimize AIC; check residual ACF (Ljung-Box test)
      \end{enumerate}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{examplebox}{US Retail Sales}
        SARIMA(1,1,1)(0,1,1)[12]
        is a strong starting point for monthly retail data
        with trend and December seasonality.
      \end{examplebox}
    \end{column}
  \end{columns}
\end{frame}

\section{VAR Models}
% =============================================================================

\sectionslide{VAR Models}{%
  When two series influence each other, model them together symmetrically.}

% --- Slide: Why multivariate --------------------------------------------------
\begin{frame}{Why Use Multiple Series?}
  All univariate models (ETS, ARIMA) use only $y_t$'s own history.
  But many business variables are \emph{linked}:
  \begin{itemize}
    \item Consumer sentiment today $\to$ retail spending next month
    \item Interest rates today $\to$ housing starts in 3--6 months
    \item Advertising spend today $\to$ sales over the next few weeks
  \end{itemize}
  \begin{keybox}
    If series $x_t$ contains information about $y_{t+h}$
    \emph{not already in} $y_t$'s own history, then including $x_t$
    reduces forecast error. \textbf{Granger causality} \parencite{Granger1969}
    formalizes this idea.
  \end{keybox}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{VAR:} treat all variables symmetrically ---
      each variable predicts all others.
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{ARIMAX:} one direction only ---
      $x_t$ drives $y_t$ but not vice versa.
    \end{column}
  \end{columns}
\end{frame}

% --- Slide: VAR(1) definition -------------------------------------------------
\begin{frame}{The VAR($p$) Model}
  For two stationary series $y_{1t}$ and $y_{2t}$:
  \[
    \begin{cases}
      y_{1t} = c_1 + a_{11}\,y_{1,t-1} + a_{12}\,y_{2,t-1} + \varepsilon_{1t}\\[4pt]
      y_{2t} = c_2 + a_{21}\,y_{1,t-1} + a_{22}\,y_{2,t-1} + \varepsilon_{2t}
    \end{cases}
  \]
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Key ideas:}
      \begin{itemize}\small
        \item $a_{12}$: how much $y_{2,t-1}$ predicts $y_{1t}$
              (what univariate AR misses)
        \item Each equation estimated by OLS (same regressors)
        \item Order $p$ chosen by BIC
      \end{itemize}
      \vspace{0.1cm}
      \textbf{VAR generalizes:} if $a_{12}=a_{21}=0$, each equation
      reduces to a separate AR model.
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{warningbox}
        \textbf{Parameter explosion:}
        VAR($p$) with $k$ variables has $k^2 p$ coefficients.\\[3pt]
        $k=5$, $p=4$: \textbf{100 coefficients}!\\[3pt]
        Keep $k \leq 4$ unless using LASSO-VAR.
      \end{warningbox}
    \end{column}
  \end{columns}
  \begin{keybox}
    \textbf{Python:} \texttt{statsmodels.tsa.vector\_ar.var\_model.VAR}\\
    Use \texttt{var.fit(maxlags=p, ic=`bic')} for automatic order selection.
  \end{keybox}
\end{frame}

% --- Slide: Granger causality -------------------------------------------------
\begin{frame}{Granger Causality Test}
  \begin{definitionbox}{Granger Causality}
    $x_t$ \textbf{Granger-causes} $y_t$ if past values of $x_t$ help predict
    $y_t$ \emph{beyond what $y_t$'s own past explains}.
  \end{definitionbox}
  \textbf{Test procedure:}
  \begin{enumerate}\small
    \item Fit VAR with and without $x$-lags in the $y$-equation
    \item F-test $H_0$: all $x$-lag coefficients $= 0$
    \item Reject $\Rightarrow$ $x_t$ Granger-causes $y_t$
  \end{enumerate}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Business example:}\\
      Does consumer sentiment (UMCSENT)
      Granger-cause retail sales (RSXFS)?\\
      $H_0$: all sentiment lags $= 0$ in the retail equation.
    \end{column}
    \begin{column}{0.41\textwidth}
      \begin{warningbox}
        Granger causality $\neq$ structural causality.
        It tests \emph{predictive content},
        not whether $x_t$ mechanically causes $y_t$.
      \end{warningbox}
    \end{column}
  \end{columns}
  \muted{\footnotesize\itshape
    Impulse response functions (IRFs) trace how a shock to one variable
    propagates through the VAR system over time. Useful for understanding
    the dynamics, not just the forecasts.}
\end{frame}

% --- Slide: ARIMAX -----------------------------------------------------------
\begin{frame}{ARIMAX: Exogenous Variables with ARIMA Errors}
  When one variable clearly \emph{drives} another (not symmetric), use ARIMAX:
  \begin{definitionbox}{ARIMAX Model}
    \[
      y_t = \mathbf{x}_t'\boldsymbol{\beta} + \eta_t,
      \qquad \eta_t \sim \mathrm{ARIMA}(p,d,q)
    \]
    $\mathbf{x}_t$ are exogenous regressors;
    $\eta_t$ captures remaining autocorrelation in the residuals.
  \end{definitionbox}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{When to use ARIMAX:}
      \begin{itemize}\small
        \item Causal direction is clear: ads $\to$ sales
        \item You already have or can forecast $x_t$
        \item Adding $x_t$ reduces forecast error in CV
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{keybox}
        \small
        \textbf{Python:}\\
        \texttt{SARIMAX(y, exog=X,}\\
        \texttt{  order=(p,d,q)).fit()}
      \end{keybox}
      \begin{warningbox}
        \small ARIMAX requires a future forecast of $x_t$ to forecast $y_t$.
        If that is unavailable, use VAR instead.
      \end{warningbox}
    \end{column}
  \end{columns}
\end{frame}

\section{Key Takeaways and Roadmap}
% =============================================================================

\sectionslide{Key Takeaways and Roadmap}{%
  Stationarity, ARIMA, VAR, and ARIMAX form the classical time series toolkit.}

% --- Slide: Key Takeaways -----------------------------------------------
\begin{frame}{Key Takeaways}
  \begin{keybox}
    {\small
    \begin{enumerate}
      \item \textbf{Stationarity} is required before fitting time series models.
            Test with ADF ($H_0$: unit root) and KPSS ($H_0$: stationary).
            First-differencing removes a stochastic trend.

      \item \textbf{ACF} identifies MA order; \textbf{PACF} identifies AR order.
            Always inspect after differencing, not before.

      \item \textbf{ARIMA($p$,$d$,$q$)} combines differencing with AR/MA.
            SARIMA adds seasonal polynomials for periodic autocorrelation.
            Use \texttt{auto\_arima} for automatic selection.

      \item \textbf{VAR} models all variables symmetrically and captures
            cross-variable dynamics. Keep $k \leq 4$.
            Use \textbf{Granger causality} to test predictive content.

      \item \textbf{ARIMAX} adds exogenous predictors to ARIMA.
            Use when causal direction is clear and $x_t$ is forecastable.
    \end{enumerate}
    }
  \end{keybox}
  \muted{\itshape Next: Generalized Additive Models (GAMs) ---
  flexible, interpretable nonlinear forecasting with Prophet and pyGAM.}
\end{frame}
