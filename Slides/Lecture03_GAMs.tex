\section{Motivation: Beyond Linear, Without Black Boxes}
% =============================================================================

\sectionslide{Motivation: Beyond Linear, Without Black Boxes}{%
  We want flexibility to capture nonlinear relationships
  while still being able to \emph{explain} what the model is doing.}

% --- Slide: The linear / nonlinear tradeoff -----------------------------------
\begin{frame}{The Flexibility--Interpretability Tradeoff}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Linear models (ARIMA, regression):}
      {\setbeamertemplate{itemize item}{\pos{$\checkmark$}}
      \begin{itemize}
        \item Easy to interpret: each coefficient
              has a clear meaning
        \item Stable, well-understood properties
      \end{itemize}}
      {\setbeamertemplate{itemize item}{\negc{$\times$}}
      \begin{itemize}
        \item Assume a constant slope everywhere
        \item Cannot capture nonlinear patterns
        \item Miss threshold effects and curvature
      \end{itemize}}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Fully nonlinear models (trees, neural nets):}
      {\setbeamertemplate{itemize item}{\pos{$\checkmark$}}
      \begin{itemize}
        \item Flexible --- can capture almost any pattern
        \item High accuracy on complex data
      \end{itemize}}
      {\setbeamertemplate{itemize item}{\negc{$\times$}}
      \begin{itemize}
        \item Hard to interpret: why did the model
              make this prediction?
        \item Interactions entangle all variables
      \end{itemize}}
    \end{column}
  \end{columns}
  \vspace{0.3cm}
  \begin{keybox}
    \textbf{GAMs offer a middle ground:} nonlinear, flexible relationships
    for each variable, but the model stays \textbf{additive} ---
    so each variable's effect can be visualized and explained independently.
  \end{keybox}
\end{frame}

% --- Slide: The nonlinear marginal effect problem ----------------------------
\begin{frame}{Why Additivity Matters for Interpretation}
  In a non-additive model, the effect of one variable depends on the value
  of another --- you cannot describe them separately.
  In a GAM, each variable's effect is isolated, making the model
  far easier to explain.

  \begin{warningbox}
    \small \textbf{Optional math detail:}
    Suppose we fit $y = x_1^2 x_2^2 + \varepsilon$.
    The marginal effect of $x_1$ is
    $\frac{\partial y}{\partial x_1} = 2x_1 x_2^2$,
    which depends on \emph{both} $x_1$ and $x_2$ ---
    you cannot describe the effect of $x_1$ without knowing $x_2$.
  \end{warningbox}

  This depends on \emph{both} $x_1$ and $x_2$ --- you cannot describe
  the effect of $x_1$ without knowing $x_2$.
  \vspace{0.2cm}
  \begin{examplebox}{GAM Marginal Effects Are Simple}
    In a GAM: $y = f_1(x_1) + f_2(x_2) + \varepsilon$

    The marginal effect of $x_1$ is just $f_1'(x_1)$ ---
    a function of $x_1$ alone, regardless of $x_2$.\\[4pt]
    \textbf{You can plot the effect of each variable separately
    and explain it to a business stakeholder.}
  \end{examplebox}
\end{frame}

\section{GAM Structure and Smoothing}
% =============================================================================

\sectionslide{GAM Structure and Smoothing}{%
  Additive nonlinear functions, penalized for wiggliness.}

% --- Slide: GAM definition ---------------------------------------------------
\begin{frame}{What Is a GAM?}
  \begin{definitionbox}{Generalized Additive Model}
    \vspace{-4pt}
    \[
      y = \alpha + f_1(x_1) + f_2(x_2) + \cdots + f_k(x_k) + \varepsilon
    \]
    \vspace{-6pt}
    \small where each $f_j(\cdot)$ is a \textbf{smooth, nonlinear function}
    estimated from the data \parencite{Wood2017}.
  \end{definitionbox}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Key properties:}
      \begin{itemize}\footnotesize
        \item Each $f_j(\cdot)$ is flexible --- linear, curved,
              U-shaped, or threshold-like
        \item Additive --- effects are \emph{independent}
              and easy to visualize
        \item Each function minimizes prediction error
              subject to a smoothness penalty
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{examplebox}{Business Application}
        \small Modeling retail sales with:
        \begin{itemize}\footnotesize
          \item $f_1(\text{time})$: long-run trend
          \item $f_2(\text{month})$: seasonal effect
          \item $f_3(\text{unemployment})$: economic sensitivity
        \end{itemize}
        Each effect plotted and interpreted separately.
      \end{examplebox}
    \end{column}
  \end{columns}
\end{frame}

% --- Slide: Smoothness penalty -----------------------------------------------
\begin{frame}{Controlling Smoothness: The Penalty Term}
  A GAM minimizes a \textbf{penalized} sum of squared errors:
  \[
    \text{PSSE} = \underbrace{\sum_{i=1}^{n}(y_i - \hat{y}_i)^2}_{\text{fit}} +
                  \underbrace{\lambda \int (f''(x))^2\,dx}_{\text{smoothness penalty}}
  \]
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{The penalty term:}
      \begin{itemize}\small
        \item $f''(x)$ is the second derivative of $f$ ---
              measures how fast the slope is changing
        \item A wiggly function has a large integral;
              a smooth function has a small integral
        \item $\lambda$ controls the trade-off:
              large $\lambda$ = smoother; small $\lambda$ = more flexible
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{keybox}
        {\small
        $\lambda \to 0$: no penalty, fit as wiggly as needed --- risk of overfitting\\[3pt]
        $\lambda \to \infty$: all $f_j$ become linear --- reduces to OLS\\[3pt]
        Optimal $\lambda$: chosen by \textbf{cross-validation} or GCV
        }
      \end{keybox}
    \end{column}
  \end{columns}
  \muted{\footnotesize \textbf{Splines} represent $f(x)$:
  piecewise polynomials that connect smoothly at \textbf{knot} points.}
\end{frame}

\section{Prophet for Time Series GAMs}
% =============================================================================

\sectionslide{Prophet for Time Series GAMs}{%
  Facebook's Prophet decomposes a time series into trend, seasonality,
  and holiday effects using a GAM.}

% --- Slide: Prophet overview -------------------------------------------------
\begin{frame}{Prophet: Forecasting at Scale}
  \textcite{TaylorLetham2018} developed Prophet for large-scale forecasting
  at Meta (Facebook). It models a time series as:
  \[
    y(t) = \underbrace{g(t)}_{\text{trend}} + \underbrace{s(t)}_{\text{seasonality}}
          + \underbrace{h(t)}_{\text{holidays}} + \varepsilon_t
  \]
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Key components:}
      \begin{itemize}\small
        \item \textbf{Trend $g(t)$:} piecewise linear or logistic growth,
              with \emph{changepoints} where the trend shifts
        \item \textbf{Seasonality $s(t)$:} Fourier series to capture
              weekly, monthly, yearly patterns
        \item \textbf{Holidays $h(t)$:} user-specified irregular events
              (Black Friday, COVID lockdowns)
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{keybox}
        Prophet is designed to be:
        \begin{itemize}\small
          \item Easy to use (one fit + forecast call)
          \item Robust to missing data and outliers
          \item Interpretable: each component plotted separately
        \end{itemize}
      \end{keybox}
    \end{column}
  \end{columns}
  \muted{\footnotesize\itshape
    \texttt{changepoint\_prior\_scale} controls trend flexibility:
    high = adapts quickly (risk of overfitting); low = stable trend.}
\end{frame}

% --- Slide: Prophet code -----------------------------------------------------
\begin{frame}{Prophet: Code and Workflow}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{examplebox}{Prophet on Retail Sales}
        {\scriptsize
        \texttt{from prophet import Prophet}\\
        \texttt{df = pd.DataFrame(\{'ds': dates, 'y': sales\})}\\
        \texttt{m = Prophet(}\\
        \texttt{\ \ changepoint\_prior\_scale=0.05,}\\
        \texttt{\ \ yearly\_seasonality=True)}\\
        \texttt{m.fit(df)}\\
        \texttt{future = m.make\_future\_dataframe(}\\
        \texttt{\ \ periods=12, freq='M')}\\
        \texttt{forecast = m.predict(future)}\\
        \texttt{m.plot\_components(forecast)}
        }
      \end{examplebox}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{Component plots:}
      \begin{itemize}\footnotesize
        \item \textbf{Trend:} long-run direction with changepoints
        \item \textbf{Yearly seasonality:} seasonal wave across months
        \item \textbf{Weekly seasonality:} day-of-week pattern
      \end{itemize}
      \textbf{When to use:}
      {\setbeamertemplate{itemize item}{\pos{$\checkmark$}}
      \begin{itemize}\footnotesize
        \item Daily/weekly/monthly series with seasonality
        \item Known holiday effects; interpretable plots needed
      \end{itemize}}
      {\setbeamertemplate{itemize item}{\negc{$\times$}}
      \begin{itemize}\footnotesize
        \item Many correlated external predictors (use pyGAM)
      \end{itemize}}
    \end{column}
  \end{columns}
\end{frame}

\section{pyGAM for General Additive Modeling}
% =============================================================================

\sectionslide{pyGAM for General Additive Modeling}{%
  When you have multiple predictors and want flexible nonlinear effects
  for each.}

% --- Slide: pyGAM overview ---------------------------------------------------
\begin{frame}{pyGAM: Flexible Additive Modeling}
  \texttt{pyGAM} fits a GAM with user-specified smooth functions for each predictor:
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{examplebox}{pyGAM Example}
        {\footnotesize
        \texttt{from pygam import LinearGAM, s, f, l}\\[2pt]
        \texttt{\# s()=spline\ \ l()=linear\ \ f()=factor}\\[2pt]
        \texttt{gam = LinearGAM(}\\
        \texttt{\ \ s(0) + s(1) + s(2) + f(3))}\\
        \texttt{gam = gam.gridsearch(X, y)}\\[2pt]
        \texttt{pdep, confi = gam.partial\_dependence(}\\
        \texttt{\ \ term=0, width=0.95)}\\[2pt]
        \texttt{y\_hat = gam.predict(X\_new)}
        }
      \end{examplebox}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{Function types:}
      \begin{description}\small
        \item[\texttt{s(j)}] Smooth (spline) function of feature $j$
              --- flexible nonlinear effect
        \item[\texttt{l(j)}] Linear function of feature $j$
        \item[\texttt{f(j)}] Factor (binary / categorical) feature $j$
      \end{description}
      \vspace{0.2cm}
      \textbf{\texttt{gridsearch}:} automatically searches for the optimal
      smoothness penalty $\lambda$ using cross-validation.
    \end{column}
  \end{columns}
\end{frame}

% --- Slide: Partial dependence -----------------------------------------------
\begin{frame}{Interpreting GAM Results: Partial Dependence Plots}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Partial dependence plot for each predictor $x_j$:}
      \begin{itemize}\small
        \item Shows the estimated function $\hat{f}_j(x_j)$ over the range
              of observed values
        \item Confidence bands show uncertainty around the smooth effect
        \item \textbf{Flat line:} $x_j$ has no effect on $y$
        \item \textbf{Increasing curve:} $x_j$ has a positive but potentially
              nonlinear effect
        \item \textbf{U-shaped:} $x_j$ has an optimal level (too little or too
              much is bad)
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{examplebox}{World Happiness Example}
        pyGAM partial dependence for
        ``freedom'' predicting happiness:
        \begin{itemize}\small
          \item Near-linear effect: more freedom $\to$ more happiness
          \item Confidence band widens at extremes (sparse data)
          \item ``Economy'' effect: clearly nonlinear --- diminishing returns
                at high income levels
        \end{itemize}
      \end{examplebox}
    \end{column}
  \end{columns}
  \begin{keybox}
    Partial dependence plots make GAM results understandable to
    business stakeholders: \emph{``As unemployment increases from 5\% to 10\%,
    predicted sales decline by X\%.''}
  \end{keybox}
\end{frame}

\section{Key Takeaways and Roadmap}
% =============================================================================

\sectionslide{Key Takeaways and Roadmap}{%
  GAMs: flexible, interpretable, and underutilized in business forecasting.}

% --- Slide: GAMs vs. alternatives --------------------------------------------
\begin{frame}{GAMs vs.\ Other Methods: When to Choose Each}
  \begin{center}
    {\small
    \begin{tabular}{lllll}
      \toprule
      \textbf{Method} & \textbf{Nonlinear} & \textbf{Interactions} & \textbf{Interpretable} & \textbf{Use case} \\
      \midrule
      ARIMA / ETS & No & No & High & Univariate TS \\
      Linear regression & No & Manual & High & Few predictors \\
      \textbf{GAM} & Yes & No & High & Few predictors, explain effects \\
      Random Forest & Yes & Yes & Medium & Many predictors, tabular \\
      XGBoost & Yes & Yes & Low & Best accuracy, tabular \\
      \bottomrule
    \end{tabular}
    }
  \end{center}
  \vspace{0.2cm}
  \begin{keybox}
    Choose GAMs when:
    \begin{itemize}\small
      \item You need to \textbf{explain} the relationship between each predictor and the outcome
      \item Predictors have nonlinear, curved effects
      \item Time series has complex trend and seasonality patterns (use Prophet)
      \item Interpretability is more important than absolute prediction accuracy
    \end{itemize}
  \end{keybox}
\end{frame}

% --- Slide: Key takeaways ---------------------------------------------------
\begin{frame}{Lecture 3: Key Takeaways}
  \begin{keybox}
    {\small
    \begin{enumerate}
      \item \textbf{GAMs} offer nonlinear flexibility while maintaining interpretability:
            $y = \alpha + f_1(x_1) + f_2(x_2) + \cdots + f_k(x_k) + \varepsilon$.
            Because the model is additive, each variable's effect can be visualized
            and explained independently.

      \item \textbf{Smoothness penalty} (PSSE) prevents overfitting.
            The penalty $\lambda$ controls the trade-off between fit and smoothness.
            It is chosen by cross-validation.

      \item \textbf{Prophet} decomposes a time series into trend, seasonality, and
            holiday effects using a GAM. Ideal for business time series with
            seasonal patterns and known irregular events.

      \item \textbf{pyGAM} fits a general GAM with any combination of predictors.
            Use \texttt{s()} for smooth effects, \texttt{l()} for linear,
            \texttt{f()} for categorical. \texttt{gridsearch} tunes $\lambda$ automatically.

      \item Partial dependence plots make results \textbf{actionable for stakeholders}:
            show clearly how each variable affects the forecast.
    \end{enumerate}
    }
  \end{keybox}
  \muted{\itshape Next: Decision Trees --- moving to tree-based methods
  that capture interactions automatically.}
\end{frame}
