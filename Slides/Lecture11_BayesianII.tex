\section{Bayesian Time Series: Trend and Seasonality}
% =============================================================================

\sectionslide{Bayesian Time Series: Trend and Seasonality}{%
  Encode structural knowledge about trend and seasonality directly
  in the model, with uncertainty estimates on each component.}

% --- Slide: Why Bayesian TS? ------------------------------------------------
\begin{frame}{From Frequentist to Bayesian Time Series}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{What ARIMA and ETS give you:}
      \begin{itemize}\footnotesize
        \item Point forecasts with asymptotic prediction intervals
        \item Implicit assumptions about trend and error structure
        \item Limited ability to incorporate domain knowledge
        \item Single-equation, single-series models
      \end{itemize}
      \vspace{0.1cm}
      \textbf{What Bayesian TS adds:}
      \begin{itemize}\footnotesize
        \item Full posterior distributions over trend, seasonality,
              and each forecast horizon
        \item Explicit priors to incorporate business knowledge
              (``growth rate is unlikely to exceed 20\%'')
        \item Natural uncertainty propagation through time
        \item Hierarchical structure for multiple related series
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{examplebox}{Business Application}
        \footnotesize A retail chain has 500 stores. Rather than fitting
        500 independent ARIMA models, a Bayesian hierarchical
        model shares information across stores:
        \begin{itemize}\footnotesize
          \item Each store gets its own parameters
          \item Parameters drawn from a common
                distribution (the hierarchy)
          \item Sparse stores borrow strength
                from data-rich stores
        \end{itemize}
      \end{examplebox}
    \end{column}
  \end{columns}
\end{frame}

% --- Slide: Bayesian local linear trend model --------------------------------
\begin{frame}{Bayesian Structural Time Series}
  \small We decompose a series into \textbf{trend}, \textbf{seasonality}, and \textbf{noise},
  each with its own priors --- giving simultaneous uncertainty estimates on all components.
  \[
    y_t = \mu_t + s_t + \varepsilon_t, \quad \varepsilon_t \sim \mathcal{N}(0, \sigma^2)
  \]
  \vspace{-0.3cm}
  where the trend $\mu_t$ evolves as:
  $\mu_t = \mu_{t-1} + \delta_{t-1} + \eta_t$, \quad $\delta_t = \delta_{t-1} + \zeta_t$
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Seasonal component (Fourier features):}

      \small Seasonality as smooth sine/cosine waves; more waves ($K$) = more flexible.
      \[
        s_t = \sum_{k=1}^{K} \left[a_k \cos\!\left(\tfrac{2\pi k t}{P}\right) + b_k \sin\!\left(\tfrac{2\pi k t}{P}\right)\right]
      \]
      \footnotesize $P$ = period (12 monthly, 4 quarterly); $K$ controls flexibility.
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{definitionbox}{Local Linear Trend}
        \small
        $\mu_t$: level at time $t$\\
        $\delta_t$: slope (trend growth rate)\\
        $\eta_t \sim \mathcal{N}(0, \sigma_\eta^2)$: level shock\\
        $\zeta_t \sim \mathcal{N}(0, \sigma_\zeta^2)$: slope shock\\[2pt]
        Priors on $\sigma_\eta, \sigma_\zeta$ control how much
        the trend can change over time.
      \end{definitionbox}
    \end{column}
  \end{columns}
\end{frame}

% --- Slide: PyMC Bayesian TS -------------------------------------------------
\begin{frame}{Bayesian TS in PyMC}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{examplebox}{Linear Trend + Seasonality}
        {\tiny
        \texttt{with pm.Model() as ts\_model:}\\
        \texttt{  alpha = pm.Normal('alpha',}\\
        \texttt{    mu=y.mean(), sigma=y.std())}\\
        \texttt{  beta = pm.Normal('beta', 0, 1)}\\[1pt]
        \texttt{  t = np.arange(len(y))}\\
        \texttt{  fourier = [pm.Normal(f'a\{k\}',0,1)*}\\
        \texttt{    np.cos(2*np.pi*k*t/12) +}\\
        \texttt{    pm.Normal(f'b\{k\}',0,1)*}\\
        \texttt{    np.sin(2*np.pi*k*t/12)}\\
        \texttt{    for k in range(1,4)]}\\
        \texttt{  mu = alpha + beta*t + sum(fourier)}\\
        \texttt{  sigma = pm.HalfNormal('sigma', y.std())}\\
        \texttt{  obs = pm.Normal('obs', mu, sigma,}\\
        \texttt{    observed=y)}\\
        \texttt{  trace = pm.sample(2000)}
        }
      \end{examplebox}
    \end{column}
    \begin{column}{0.41\textwidth}
      \textbf{What you get:}
      \begin{itemize}\footnotesize\itemsep0pt
        \item Posterior over slope $\beta$:
              growing or declining trend?
        \item Posterior over $a_k, b_k$:
              shape and strength of seasonality
        \item Posterior predictive: credible bands
              at each forecast horizon
      \end{itemize}
      \begin{keybox}
        \footnotesize \texttt{pm.sample\_posterior\_predictive(trace)}
        generates forecast draws.
        Summarize: median + 90\% credible band.
      \end{keybox}
    \end{column}
  \end{columns}
\end{frame}

\section{Hierarchical Models: Partial Pooling}
% =============================================================================

\sectionslide{Hierarchical Models: Partial Pooling}{%
  Share information across groups via a common prior distribution.
  The hierarchy learns what is typical; each group learns its own deviation.}

% --- Slide: The pooling problem ----------------------------------------------
\begin{frame}{Three Approaches to Multi-Group Data}
  Suppose we have sales data for $J = 50$ stores, each with limited history.
  \begin{columns}[T]
    \begin{column}{0.33\textwidth}
      \textbf{Complete pooling:}
      Fit one model ignoring stores.
      {\setbeamertemplate{itemize item}{\negc{$\times$}}
      \begin{itemize}\small
        \item Assumes all stores are identical
        \item Misses systematic store differences
        \item Wrong predictions for outlier stores
      \end{itemize}}
    \end{column}
    \begin{column}{0.33\textwidth}
      \textbf{No pooling:}
      Fit 50 independent models.
      {\setbeamertemplate{itemize item}{\negc{$\times$}}
      \begin{itemize}\small
        \item Ignores shared structure
        \item Noisy estimates for stores with few observations
        \item Overfits sparse data stores
      \end{itemize}}
    \end{column}
    \begin{column}{0.33\textwidth}
      \textbf{Partial pooling (hierarchical):}
      Store params drawn from a common prior.
      {\setbeamertemplate{itemize item}{\pos{$\checkmark$}}
      \begin{itemize}\small
        \item Each store has its own estimate
        \item Sparse stores shrink toward the group mean
        \item Data-rich stores less affected by prior
      \end{itemize}}
    \end{column}
  \end{columns}
  \vspace{0.2cm}
  \begin{keybox}
    Partial pooling is the Bayesian answer to the bias-variance trade-off:
    adding a hierarchical prior reduces variance (less overfitting) at the cost
    of a small bias (shrinkage toward the mean).
    \parencite{Gelman2013}
  \end{keybox}
\end{frame}

% --- Slide: Hierarchical model -----------------------------------------------
\begin{frame}{Hierarchical Model: Structure and Code}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \small\textbf{Partial pooling for store-level intercepts:}
      \vspace{-0.2cm}
      {\small
      \begin{align*}
        y_{it} &\sim \mathcal{N}(\alpha_i + \beta t, \sigma^2) \\
        \alpha_i &\sim \mathcal{N}(\mu_\alpha, \sigma_\alpha^2), \quad
        \mu_\alpha \sim \mathcal{N}(\bar{y}, \text{SD}(y)), \quad
        \sigma_\alpha \sim \text{HN}(1)
      \end{align*}
      }
      \vspace{-0.2cm}
      \begin{examplebox}{PyMC Partial Pooling}
        {\footnotesize
        \texttt{with pm.Model() as hier:}\\
        \texttt{  mu\_a  = pm.Normal('mu\_a', 0, 1)}\\
        \texttt{  sig\_a = pm.HalfNormal('sig\_a', 1)}\\
        \texttt{  alpha = pm.Normal('alpha',}\\
        \texttt{    mu=mu\_a, sigma=sig\_a, shape=J)}\\
        \texttt{  beta = pm.Normal('beta', 0, 1)}\\
        \texttt{  mu = alpha[store\_idx] + beta * t}\\
        \texttt{  pm.Normal('obs', mu, 1, observed=y)}
        }
      \end{examplebox}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{keybox}
        \small\textbf{Key:} each store gets $\alpha_i$, all drawn from
        $\mathcal{N}(\mu_\alpha, \sigma_\alpha)$.
        Estimating $\mu_\alpha, \sigma_\alpha$ from data
        is what makes the model hierarchical.
      \end{keybox}
      \vspace{0.1cm}
      \begin{warningbox}
        \small \textbf{Non-centered parameterization}
        improves sampling:
        $\alpha_i = \mu_\alpha + \sigma_\alpha z_i$,
        $z_i \sim \mathcal{N}(0,1)$.
      \end{warningbox}
    \end{column}
  \end{columns}
\end{frame}

\section{Key Takeaways}
% =============================================================================

\sectionslide{Key Takeaways}{%
  Bayesian TS and hierarchical models: full uncertainty,
  shared structure, better predictions for sparse groups.}

% --- Slide: Key takeaways ---------------------------------------------------
\begin{frame}{Lecture 11: Key Takeaways}
  \begin{keybox}
    {\small
    \begin{enumerate}
      \item Bayesian time series models decompose the series into
            \textbf{trend}, \textbf{seasonality}, and \textbf{error},
            each with its own prior. The posterior quantifies uncertainty
            in every component simultaneously.

      \item \textbf{Fourier features} ($\cos$ and $\sin$ terms) parameterize
            seasonality with $K$ harmonics. More harmonics = more flexible
            seasonal patterns.

      \item \textbf{Hierarchical models} use partial pooling to share
            information across groups (stores, regions, products).
            Sparse groups shrink toward the group mean; data-rich groups
            are less affected.

      \item The three pooling options --- complete, no, and partial ---
            represent a bias-variance trade-off. Partial pooling
            (hierarchical) is almost always the best choice for
            multi-group business data.

      \item In PyMC, hierarchical structure is created by giving
            group-level parameters a shared prior with hyperparameters
            that are themselves estimated from the data.
    \end{enumerate}
    }
  \end{keybox}

  posterior coefficient distributions, and conditional analysis.}
\end{frame}
